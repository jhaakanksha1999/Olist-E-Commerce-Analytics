-- RFM analysis 
Create Table rfm_table as
select customer_unique_id,
datediff((select max(order_purchase_timestamp) from olist_data),Max(order_purchase_timestamp)) as recency,
count(distinct order_id) as frequency,
round(sum(payment_value),2) as monetary
from olist_data
group by customer_unique_id;

select * from rfm_table where frequency >1;

 create table segment_table as 
 with rfm_score as (select * ,
NTILE(5) over(order by recency desc) as r_score,
NTILE(5) over(order by frequency asc) as f_score,
NTILE(5) over(order by monetary asc) as m_score
from rfm_table)
select*,
case 
when r_score>= 4 and f_score>= 4 then 'Loyal Customer'
when r_score>= 4 and f_score <= 2 then 'New Customer'
when r_score <= 4 and f_score>= 4 then 'At risk'
else 'Others'
end as segment from rfm_score;

select * from segment_table;

-- We define churn simply as customer who have not purchased in 90 days

with churn_table as(
select * ,
case when recency > 180 then 'churned'
else 'Active' end as churn_status
from rfm_table)
select churn_status,
count(*) as customers
from churn_table
group by churn_status;

-- churn by delivery delay
select

-- Funnel analysis, highlighting gaps between order approval and final delivery 
create table funnel_table as 
select order_id,customer_unique_id,
count(distinct order_id) as total_order,
count(distinct case when order_status in ('approved','shipped','delivered') then order_id end ) as approved_orders,
count(distinct case when order_status="delivered" then order_id end ) as delivered_orders,
count(distinct case when order_status="delivered" and  review_score is not null then order_id end) as reviewed_orders
from olist_data
group by order_id,customer_unique_id;

select * from funnel_table;

select 
count(distinct order_id) as total_orders,
(count(distinct case when order_status="delivered" then order_id end)*1.0/count(distinct order_id) )as delivery_rate,
count(distinct case when order_status = "delivered" and review_score is not null then order_id end)*1.0/count(distinct order_id ) as review_rate
from olist_data;

SELECT 
    r.segment,
    COUNT(DISTINCT f.order_id) AS total_orders
FROM funnel_table f
JOIN segment_table r
ON f.customer_unique_id = r.customer_unique_id
GROUP BY r.segment;

 -- creating views for Power BI Dashboard
create view vw_cust_segment as
with rfm_base as(
select customer_unique_id,
datediff(
(select max(order_purchase_timestamp) from olist_data),max(order_purchase_timestamp)) as recency,
count(distinct order_id) as frequency,
round(sum(payment_value),2) as monetary
from olist_data 
group by customer_unique_id),
rfm_score as(
select *,
NTILE(5) over(order by recency desc) as r_score,
NTILE(5) over(order by frequency asc) as f_score,
NTILE(5) over(order by monetary asc) as m_score
from rfm_base)
select * ,
case when r_score >= 4 and f_score >= 4 then "Loyal"
when r_score >= 4 and f_score <= 2 then "New"
when r_score <= 2 and f_score >= 4 then "At Risk"
else "Other" end as segment,
 CASE 
WHEN recency > 180 THEN 'Churned'
ELSE 'Active'
END AS churn_status
FROM rfm_score;

select 
case when delivery_time <= 7 then "1.Fast"
when delivery_time <= 14 then "2. Normal(8-14 days)"
when delivery_time <= 21 then "3. Slow(15-21 days)"
when delivery_time >= 21 then "4. Very Slow(21+ days)"
else "Unknown" end as delivery_bucket,
round(avg(review_score),2) as avg_review_score,
count(distinct order_id) as total_order
from olist_data
where delivery_time is not null
and review_score is not null 
group by delivery_bucket
order by delivery_bucket
